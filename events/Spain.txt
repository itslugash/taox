country_event = { # "Civil War!"
	id = spain.1
	title = spain.1.t
	desc = spain.1.d
	picture = GFX_report_event_spain_civil_war_soldiers
	
	trigger = {
		NOT = {
			has_global_flag = scw_over
		}
		tag = SPR
		has_defensive_war = no
		NOT = {
			has_government = right-wing
		}
		date > 1936.2.16
	}

	mean_time_to_happen = {
		days = 151
	}

	fire_only_once = yes

	option = { # "The Nationalists!"
		name = spain.1.a
		trigger = {
			is_ai = no
		}
		hidden_effect = {
			set_capital = 176
			start_civil_war = {
				ruling_party = right-wing
				ideology = left-wing
				size = 0.5
				capital = 41
				states = {
					41 165 167 168 172 173 175
				}
			}
			random_other_country = {
				limit = {
					original_tag = SPR
					has_government = left-wing
				}
				load_oob = "SPR_Civil_War"
				set_equipment_fraction = 0.5
				set_stability = 0.9
				set_war_support = 0.8
				country_event = {
					id = spain.3
					days = 1
				}
				add_popularity = {
					ideology = centre
					popularity = 0.35
				}
			}
			news_event = { # "The Spanish Civil War"
				id = news.63
				hours = 6
			}
			set_global_flag = spanish_civil_war
			
			load_oob = "SPA_Civil_War"
			set_equipment_fraction = 0.5
			set_stability = 0.9
			set_war_support = 0.8
			
			add_popularity = {
				ideology = right-wing
				popularity = 1
			}
			random_other_country = {
				limit = {
					original_tag = SPR
					has_government = left-wing
				}
				set_cosmetic_tag = SPR_civil_war
			}

			country_event = { # "Seek International Support?"
				id = spain.2
				days = 7
			}
		}
		custom_effect_tooltip = nationalists_chosen
	}
	option = { # "The Republicans!"
		name = spain.1.b
		hidden_effect = {
			start_civil_war = {
				ideology = right-wing
				size = 0.5
				capital = 176
				states = {
					166 699 178 290 169 170 176 174 171 177 297
				}
			}
			random_other_country = {
				limit = {
					original_tag = SPR
					has_government = right-wing
				}
				set_equipment_fraction = 0.5
				set_stability = 0.9
				set_war_support = 0.8
				country_event = { # # "Seek International Support?"
					id = spain.2
					days = 1
				}
				load_oob = "SPA_Civil_War"
			}
			news_event = { # "The Spanish Civil War"
				id = news.63
				hours = 6
			}
			set_global_flag = spanish_civil_war
			
			set_politics = {
				ruling_party = left-wing
				elections_allowed = no
			}
			add_popularity = {
				ideology = left-wing
				popularity = 0.5
			}
			add_popularity = {
				ideology = right-wing
				popularity = -1
			}	
			random_other_country = {
				limit = {
					original_tag = SPR
					has_government = right-wing
				}
				set_cosmetic_tag = SPR_civil_war
			}
			load_oob = "SPR_Civil_War"
			set_equipment_fraction = 0.5
			set_stability = 0.9
			set_war_support = 0.8

				days = 7

country_event = { # "Seek International Support?"
	id = spain.2
	title = spain.2.t
	desc = spain.2.d
	picture = GFX_report_event_spain_civil_war_volunteers_01
	
	is_triggered_only = yes

	trigger = {
		has_civil_war = yes
		original_tag = SPR
	}

	option = { # "Let's request equipment and munitions."
		name = spain.2.a
		every_other_country = {
			limit = {
				has_government = right-wing
				NOT = {
					original_tag = SPR
				}
			}
			country_event = { # "Nationalist Spain Wants Our Help"
				id = spain.4
				days = 1
			}
		}
		if = {
			limit = {
				POR = {
					NOT = {
						has_government = left-wing
					}
				}
			}
			POR = {
				country_event = { # "Nationalist Spain Wants Our Help"
					id = spain.4
					days = 1
				}
			}
		}
	}

	option = { # "Spain must be won with Spanish blood!"
		name = spain.2.b
		add_political_power = 40
		ai_chance = {
			factor = 0
		}
	}
}
