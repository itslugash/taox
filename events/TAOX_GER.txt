add_namespace = taox_ger

country_event = { # "The Fate of Denmark (War)"
    id = taox_ger.913
    title = taox_ger.913.t
    desc = taox_ger.913.d
    picture = GFX_report_event_election_vote

    is_triggered_only = yes

    trigger = {
    	has_country_flag = GER_DEN_occupied_war
    }

    option = {
        name = taox_ger.913.a # "Recruit local help."
        retire_country_leader = yes
       	if = {
			limit = {
				AND = {
					country_exists = DEN
					DEN = {
						is_in_faction = yes
					}
				}
			}
			hidden_effect = {
				DEN = {
					set_state_controller = 37
					set_state_controller = 99
				}
			}
			DEN = {
				start_civil_war = {
					ideology = right-wing
					size = 1
				}
			}
			random_country = {
				limit = {
					tag = DEN
					has_government = right-wing
				}
				transfer_state = 37
				transfer_state = 99
				if = {
					limit = {
						has_dlc = "Death or Dishonor"
					}
					GER = {
						set_autonomy = {
							target = DEN
							autonomy_state = autonomy_satellite
						}
					}
					else = {
						puppet = DEN
					}
				}
				set_cosmetic_tag = GER_DEN
				set_politics = {

					parties = {
						left-wing = {
							popularity = 2.4
						}

						centre = { 
							popularity = 95.8
						}

						right-wing = {
							popularity = 1.8
						}

						syncretic = { 
							popularity = 0
						}
					}
					ruling_party = centre
				}
				set_party_name = {
			        ideology = centre
			        long_name = "Unity Government"
			        name = "Unity Government"
			    }
			    if = {
			    	limit = {
			    		date < 1942.5.4
			    	}
			    	create_country_leader = {
				        name = "Thorvald Stauning"
				        desc = "DEN_Thorvald_Stauning_desc"
				        picture = "Portrait_Denmark_Thorvald_Stauning.dds"
				        expire = "1965.1.1"
				        ideology = social_democracy
				        traits = {
				        	#
				        }
				    }
				    else = {
				    	if = {
				    		limit = {
				    			AND = {
					    			date > 1942.5.3
					    			date < 1942.11.10
					    		}
				    		}
				    		create_country_leader = {
						        name = "Vilhelm Buhl"
						        desc = "DEN_Vilhelm_Buhl_desc"
						        picture = "Portrait_DEN_Vilhelm_Buhl.dds"
						        expire = "1965.1.1"
						        ideology = social_democracy
						        traits = {
						        	#
						        }
						    }
						    else = {
							    create_country_leader = {
							        name = "Erik Scavenius"
							        desc = "DEN_Erick_Scavenius_desc"
							        picture = "Portrait_DEN_Erick_Scavenius.dds"
							        expire = "1965.1.1"
							        ideology = centre_independent
							        traits = {
							        	#
							        }
							    }
						    }
				    	}
				    }
			    }
			}
		}
    }
    option = {
    	name = taox_ger.913.b # "Impose direct administration."
    	retire_country_leader = yes
    	if = {
			limit = {
				country_exists = DEN
			}
			hidden_effect = {
				DEN = {
					set_state_controller = 37
					set_state_controller = 99
				}
			}
			DEN = {
				start_civil_war = {
					ideology = right-wing
					size = 1
				}
			}
			random_country = {
				limit = {
					tag = DEN
					has_government = right-wing
				}
				transfer_state = 37
				transfer_state = 99

				GER = {
					puppet = PREV
					set_autonomy = {
						target = PREV
						autonomy_state = autonomy_reichskommisariat
					}
				}
				set_cosmetic_tag = GER_DEN_reichsprotectorate
				set_party_name = {
			        ideology = right-wing
			        long_name = "Danmarks Nationalsocialistiske Arbejderparti"
			        name = "DNSAP"
			    }
			 	create_country_leader = {
			        name = "Werner Best"
			        desc = "DEN_Frits_Clausen_desc"
			        picture = "Portrait_DEN_Frits_Clausen.dds"
			        expire = "1965.1.1"
			        ideology = fascism
			        traits = {
			        	#
			        }
			    }
			}
    }
}
